155 void mailer::checkRFCcompat() {
156  // Check the line breaks.
157  std::vector<char>::iterator it;
158  for(it = message.begin(); it != message.end(); ++it) {
159  // look for \n add \r before if not there. Pretty lame but still.
160  // haven't thought of a better way yet.
161  if(*it == '\n') {
162  if(it == message.begin()) {
163  it = message.insert(it, '\r');
164  ++it; // step past newline
165  continue;
166  }
167  if((*(it -1) != '\r') ) {
168  // add a return before '\n'
169  it = message.insert(it, '\r');
170  ++it; // step past newline
171  }
172  }
173  }
174 
175  // if we get a period on a line by itself
176  // add another period to stop the server ending the mail prematurely.
177  // ( suggested by david Irwin )
178  if(message.size() == 1) {
179  if(*(message.begin()) == '.')
180  message.push_back('.');
181  }
182  else if(message.size() == 2) {
183  if(*(message.begin()) == '.') {
184  it = message.begin();
185  it = message.insert(it, '.');
186  }
187  }
188  else {
189  if(*(message.begin()) == '.') {
190  it = message.begin();
191  it = message.insert(it, '.');
192  }
193  for(it = message.begin()+2; it != message.end(); ++it) {
194  // follow the rfc. Add '.' if the first character on a line is '.'
195  if(*it == '\n') {
196  if( ((it + 1) != message.end()) && (*(it +1) == '.') ) {
197  it = message.insert(it + 1, '.');
198  ++it; // step past
199  }
200  }
201  }
202  }
203 
204  // don't do anything if we are not longer than a 1000 characters
205  if(message.size() < 1000)
206  return;
207 
208  // now we have checked line breaks
209  // check line lengths.
210  int count(1);
211  for(it = message.begin(); it < message.end(); ++it, ++count) {
212  if(*it == '\r') {
213  count = 0; // reset for a new line.
214  ++it; // get past newline
215  continue;
216  }
217  else if(count >= 998) {
218  ++it;
219  if(*it != ' ') { // we are not in a word!!
220  // it should never get to message.begin() because we
221  // start at least 998 chars into the message!
222  // Also, assume a word isn't bigger than 997 chars! (seems reasonable)
223  std::vector<char>::iterator pos = it;
224  for(int j = 0; j < 997; ++j, --pos) {
225  if(*pos == ' ') {
226  it = ++pos; // get past the space.
227  break;
228  }
229  }
230  }
231  if(it < message.end())
232  it = message.insert(it, '\r');
233  ++it;
234  if(it < message.end())
235  it = message.insert(it, '\n');
236  count = 0; // reset for a new line.
237  }
238  }
239  count=1; // reset the count
240  if(messageHTML.size()) {
241  for(it = messageHTML.begin(); it < messageHTML.end(); ++it, ++count) {
242  if(*it == '\r') {
243  count = 0; // reset for a new line.
244  ++it; // get past newline
245  continue;
246  }
247  else if(count >= 998) {
248  ++it;
249  if(*it != ' ') { // we are in a word!!
250  // it should never get to message.begin() because we
251  // start at least 998 chars into the message!
252  // Also, assume a word isn't bigger than 997 chars! (seems reasonable)
253  std::vector<char>::iterator pos = it;
254  for(int j = 0; j < 997; ++j, --pos) {
255  if(*pos == ' ') {
256  it = ++pos; // get past the space.
257  break;
258  }
259  }
260  }
261  if(it < messageHTML.end())
262  it = messageHTML.insert(it, '\r');
263  ++it;
264  if(it < messageHTML.end())
265  it = messageHTML.insert(it, '\n');
266  count = 0; // reset for a new line.
267  }
268  }
269  }

364 void mailer::operator()() {
365  returnstring = ""; // clear out any errors from previous use
366 
367  if(!recipients.size()) {
368  returnstring = "451 Requested action aborted: local error who am I mailing";
369  return;
370  }
371  if(!fromAddress.address.length()) {
372  returnstring = "451 Requested action aborted: local error who am I";
373  return;
374  }
375  if(!nameserver.length()) {
376  returnstring = "451 Requested action aborted: local error no SMTP/name server/smtp server";
377  return;
378  }
379 
380  std::vector<SOCKADDR_IN> adds;
381  if(lookupMXRecord) {
382  if(!gethostaddresses(adds)) {
383  // error!! we are dead.
384  returnstring = "451 Requested action aborted: No MX records ascertained";
385  return;
386  }
387  }
388  else { // connect directly to an SMTP server.
389  SOCKADDR_IN addr(nameserver, port, AF_INET);
390  hostent* host = 0;
391  if(addr) {
392  host = gethostbyaddr(addr.get_sin_addr(), sizeof(addr.ADDR.sin_addr), AF_INET);
393  }
394  else
395  host = gethostbyname(nameserver.c_str());
396  if(!host) {
397  returnstring = "451 Requested action aborted: local error in processing";
398  return; // error!!!
399  }
400  //memcpy(addr.get_sin_addr(), host->h_addr, host->h_length);
401  std::copy(host->h_addr_list[0], host->h_addr_list[0] + host->h_length, addr.get_sin_addr());
402  adds.push_back(addr);
403  }
404 
405  SOCKET s;
406  if(!Socket(s, AF_INET, SOCK_STREAM, 0)) {
407  returnstring = "451 Requested action aborted: socket function error";
408  return;
409  }
410 
411  if(!adds.size()) { // oops
412  returnstring = "451 Requested action aborted: No MX records ascertained";
413  }
414 
415  const std::string OK("250");
416  const std::vector<char> smtpheader(makesmtpmessage());
417  const int buffsize(1024);
418  char buff[buffsize] = "";
419 
420  for(std::vector<SOCKADDR_IN>::const_iterator address = adds.begin();
421  address < adds.end(); ++address) {
422  if(!Connect(s, *address)) {
423  returnstring = "554 Transaction failed: server connect error.";
424  continue;
425  }
426 
427  // 220 the server line returned here
428  int len1;
429  if(!Recv(len1, s, buff, buffsize -1, 0)) {
430  returnstring = "554 Transaction failed: server connect response error.";
431  continue;
432  }
433 
434  // get our hostname to pass to the smtp server
435  char hn[buffsize] = "";
436  if(gethostname(hn, buffsize)) {
437  // no local hostname!!! make one up
438  strcpy(hn, "flibbletoot");
439  }
440  std::string commandline(std::string("EHLO ") + hn + std::string("\r\n"));
441  // say hello to the server
442 
443  if(!Send(len1, s, commandline.c_str(), commandline.length(), 0)) {
444  returnstring = "554 Transaction failed: EHLO send";
445  continue;
446  }
447  if(!Recv(len1, s, buff, buffsize -1, 0)) {
448  returnstring = "554 Transaction failed: EHLO receipt";
449  continue;
450  }
451 
452  buff[len1] = '\0';
453  std::string greeting = returnstring = buff;
454  if(returnstring.substr(0,3) != OK) {
455  if(auth) {
456  // oops no ESMTP but using authentication no go bail out!
457  returnstring = "554 possibly trying to use AUTH without ESMTP server, ERROR!";
458  continue;
459  }
460  // maybe we only do non extended smtp
461  // send HELO instead.
462  commandline[0] = 'H';
463  commandline[1] = 'E';
464  if(!Send(len1, s, commandline.c_str(), commandline.length(), 0)) {
465  returnstring = "554 Transaction failed: HELO send";
466  continue;
467  }
468  if(!Recv(len1, s, buff, buffsize -1, 0)) {
469  returnstring = "554 Transaction failed: HELO receipt";
470  continue;
471  }
472  buff[len1] = '\0';
473 
474  returnstring = buff;
475  if(returnstring.substr(0,3) != OK) {
476  // we must issue a quit even on an error.
477  // in keeping with the rfc's
478  if(Send(len1, s, "QUIT\r\n", 6, 0)) {
479  char dummy[buffsize];
480  Recv(len1, s, dummy, buffsize -1, 0);
481  }
482  Closesocket(s);
483  // don't know what went wrong here if we are connected!!
484  // we continue because maybe we have more than 1 server to connect to.
485  continue;
486  }
487  }
488 
489  if(auth)
490  if(!authenticate(greeting, s))
491  continue; // try the next server, you never know!!
492 
493  // MAIL
494  // S: MAIL FROM:<Smith@Alpha.ARPA>
495  // R: 250 OK
496  // e.g. "MAIL FROM:<someone@somewhere.com>\r\n"
497  // or "MAIL FROM: John Wiggins <someone@somewhere.com>"
498  commandline = "MAIL FROM:<" + fromAddress.address + ">\r\n";
499  if(!Send(len1, s, commandline.c_str(), commandline.length(), 0)) {
500  returnstring = "554 MAIL FROM sending error";
501  continue;
502  }
503 
504  if(!Recv(len1, s, buff, buffsize -1, 0)) {
505  returnstring = "554 MAIL FROM receipt error";
506  continue;
507  }
508 
509  buff[len1] = '\0';
510  returnstring = buff;
511  if(returnstring.substr(0,3) != OK) {
512  // we must issue a quit even on an error.
513  // in keeping with the rfc's
514  if(Send(len1, s, "QUIT\r\n", 6, 0)) {
515  char dummy[buffsize];
516  Recv(len1, s, dummy, buffsize -1, 0);
517  }
518  Closesocket(s);
519  // don't know what went wrong here if we are connected!!
520  // we continue because maybe we have more than 1 server to connect to.
521  continue;
522  }
523 
524  for(recipient_const_iter recip = recipients.begin(); recip < recipients.end(); ++recip) {
525  // RCPT
526 
527  // S: RCPT TO:<Jones@Beta.ARPA>
528  // R: 250 OK
529  commandline = "RCPT TO: <" + (*recip).first.address + ">\r\n";
530  // S: RCPT TO:<Green@Beta.ARPA>
531  // R: 550 No such user here
532  //
533  // S: RCPT TO:<Brown@Beta.ARPA>
534  // R: 250 OK
535  if(!Send(len1, s, commandline.c_str(), commandline.length(), 0)) {
536  returnstring = "554 Transaction failed";
537  continue;
538  }
539  if(!Recv(len1, s, buff, buffsize -1, 0)) {
540  returnstring = buff;
541  continue;
542  }
543  buff[len1] = '\0';
544  returnstring = buff;
545  if(returnstring.substr(0,3) != OK) {
546  // This particular recipient does not exist!
547  // not strictly an error as we may have more than one recipient
548  // we should have an error vector e.g.
549  // vector<pair<string address, string error> > errs;
550  // errs.push_back(make_pair(recip->first, returnstring));
551  //
552  // we then need a function to return this vector.
553  // e.g. const vector<pair<string address, string error> >& getrecipienterrors();
554  continue;
555  }
556  }
557 
558  // DATA
559 
560  // S: DATA
561  // R: 354 Start mail input; end with <CRLF>.<CRLF>
562  // S: Blah blah blah...
563  // S: ...etc. etc. etc.
564 
565  // S: <CRLF>.<CRLF>
566  // R: 250 OK
567  if(!Send(len1, s, "DATA\r\n", 6, 0)) {
568  returnstring = "554 DATA send error";
569  continue;
570  }
571  if(!Recv(len1, s, buff, buffsize -1, 0)) {
572  returnstring = "554 DATA, server response error";
573  continue;
574  }
575  buff[len1] = '\0';
576  returnstring = buff;
577  if(returnstring.substr(0,3) != "354") {
578  // we must issue a quit even on an error.
579  // in keeping with the rfc's
580 
581  if(Send(len1, s, "QUIT\r\n", 6, 0)) {
582  char dummy[buffsize];
583  Recv(len1, s, dummy, buffsize -1, 0);
584  }
585  Closesocket(s);
586  continue;
587  }
588  // Sending the email
589  /*if(!Send(len1, s, smtpheader.c_str(), smtpheader.length(), 0)) {*/
590  if(!Send(len1, s, &smtpheader[0], smtpheader.size(), 0)) {
591  returnstring = "554 DATA, server response error (actual send)";
592  continue;
593  }
594  if(!Recv(len1, s, buff, buffsize -1, 0)) {
595  returnstring = "554 DATA, server response error (actual send)";
596  continue;
597  }
598 
599  // The server should give us a 250 reply if the mail was delivered okay
600  buff[len1] = '\0';
601  returnstring = buff;
602  if(returnstring.substr(0,3) != OK) {
603  // we must issue a quit even on an error.
604  // in keeping with the rfc's
605  if(Send(len1, s, "QUIT\r\n", 6, 0)) {
606  char dummy[buffsize];
607  Recv(len1, s, dummy, buffsize -1, 0);
608  }
609  Closesocket(s);
610  continue;
611  }
612  // hang up the connection
613  if(Send(len1, s, "QUIT\r\n", 6, 0)) {
614  char dummy[buffsize];
615  Recv(len1, s, dummy, buffsize -1, 0);
616  }
617 
618  // Let the server give us our 250 reply.
619  //buff[len1] = '\0';
620  //returnstring = buff;
621 
622  // for future reference the server is meant to give a 221 response to a quit.
623  if(returnstring.substr(0,3) != "221") {
624  // maybe will use this later
625  }
626  Closesocket(s); // disconnect
627 
628  // Let the server give us our 250 reply.
629  // don't continue as we have delivered the mail
630  // at this point just leave. all done
631  //returnstring = "250 Requested mail action okay, completed";
632  break;
633  }

636 std::vector<char> mailer::makesmtpmessage() const {
637  std::string sender(fromAddress.address);
638  if(sender.length()) {
639  std::string::size_type pos(sender.find("@"));
640  if(pos != std::string::npos) { //found the server beginning
641  sender = sender.substr(0, pos);
642  }
643  }
644 
645  std::vector<char> ret;
646  std::string headerline;
647  if(fromAddress.name.length()) {
648  headerline = "From: " + fromAddress.address + " (" + fromAddress.name + ") \r\n"
649  "Reply-To: " + fromAddress.address + "\r\n";
650  ret.insert(ret.end(), headerline.begin(), headerline.end());
651  }
652  else {
653  headerline = "From: " + fromAddress.address + "\r\n"
654  "Reply-To: " + fromAddress.address + "\r\n";
655  ret.insert(ret.end(), headerline.begin(), headerline.end());
656  }
657  headerline.clear(); // clearout our temp variable
658 
659  // add the recipients to the header
660  std::vector<std::string> to, cc, bcc;
661  for(recipient_const_iter recip = recipients.begin(); recip < recipients.end(); ++recip) {
662  if(recip->second == TO) {
663  to.push_back(recip->first.address);
664  }
665  else if(recip->second == Cc) {
666  cc.push_back(recip->first.address);
667  }
668  else if(recip->second == Bcc) {
669  bcc.push_back(recip->first.address);
670  }
671  }
672  vec_str_const_iter it; // instead of making three on the stack, just one (stops VC whining too)
673  // next section adds To: Cc: Bcc: lines to the header
674  int count = to.size();
675  if(count)
676  headerline += "To: ";
677  for(it = to.begin(); it < to.end(); ++it) {
678  headerline += *it;
679  if(count > 1 && ((it + 1) < to.end()) )
680  headerline += ",\r\n "; // must add a space after the comma
681  else
682  headerline += "\r\n";
683  }
684  count = cc.size();
