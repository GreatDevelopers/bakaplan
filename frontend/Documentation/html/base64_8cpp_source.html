170 std::vector<char> base64encode(const std::vector<char>& input, const bool returns) {
171  std::vector<char> output;
172 
173  // add a newline (SMTP demands less than 1000 characters in a message line).
174  long count = 0;
175  for(std::vector<char>::size_type p = 0; p < input.size(); p+=3) {
176  output.push_back(getbase64character((input[p] & 0xFC) >> 2));
177  ++count;
178 
179  if(p+1 < input.size()) {
180  output.push_back(getbase64character(((input[p] & 0x03) <<4) | ((input[p+1] & 0xF0) >> 4)));
181  ++count;
182  }
183  if(p+2 < input.size()) {
184  output.push_back(getbase64character(((input[p+1] & 0x0F) <<2) | ((input[p+2] & 0xC0) >>6)));
185  output.push_back(getbase64character((input[p+2] & 0x3F)));
186  ++count;
187  }
188 
189  if(p+1 == input.size()) {
190  output.push_back(getbase64character(((input[p] & 0x03) <<4)));
191  }
192  else if(p+2 == input.size()) {
193  output.push_back(getbase64character(((input[p+1] & 0x0F) <<2)));
194  }
195 
196  if(returns) {
197  // 79 characters on a line.
198  if(count > 75) {
199  output.push_back('\r');
200  output.push_back('\n');
201  count = 0;
202  }
203  }
204  }
205 
206  int pad(input.size() % 3);
207  if(pad) {
208  if(pad == 1)
209  pad = 2;
210  else
211  pad = 1;
212  }
213  for(int i = 0; i < pad; ++i)
214  output.push_back('=');
215 
216  return output;
217 }

219 std::string base64encode(const std::string& input, const bool returns) {
220  std::vector<char> in, out;

222  in.push_back(*it);

224  out = base64encode(in, returns);
